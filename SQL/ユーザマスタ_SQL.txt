DROP TABLE IF EXISTS user;



//STEP2.テーブルを作成する
//---------------------------------------------------
CREATE TABLE user(
  userid INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
  user VARCHAR(20) NOT NULL,
  name VARCHAR(20) NOT NULL,
  pw VARCHAR(10) NOT NULL
);
