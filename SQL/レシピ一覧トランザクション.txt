//STEP1.テーブルがすでに存在していれば、削除する。
//---------------------------------------------------
DROP TABLE IF EXISTS recipe;



//STEP2.テーブルを作成する
//---------------------------------------------------
CREATE TABLE recipe(
  id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
  userid INT NOT NULL,
  recipe VARCHAR(30) NOT NULL,
  cost INT NOT NULL,
  time INT,
  url VARCHAR(100),
  remarks VARCHAR(200)
);
