//記録備考トランザクション
//---------------------------------------------------


//STEP1.テーブルがすでに存在していれば、削除する。
//---------------------------------------------------
DROP TABLE IF EXISTS record_note;



//STEP2.テーブルを作成する
//---------------------------------------------------
CREATE TABLE record_note(
  id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
  userid INT NOT NULL,
  date DATE NOT NULL,
  remarks VARCHAR(200)
);
